# Docker Compose Rails App

## Create a new rails project

```bash
rails new docker-workshop-app --database=postgresql
```

### Update database.yml

```ruby
default: &default
  adapter: postgresql
  encoding: unicode
  pool: <%= ENV.fetch("RAILS_MAX_THREADS") { 5 } %>
  username: <%= ENV.fetch('DB_USERNAME') { 'user' } %>
  password: <%= ENV.fetch('DB_PASSWORD') { 'password' } %>
  host: <%= ENV.fetch('DB_HOST') { 'localhost' } %>
  port: 5432

development:
  <<: *default
  database: docker-workshop-app_development

test:
  <<: *default
  database: docker-workshop-app_test

production:
  <<: *default
```

## Create a Dockerfile
```
FROM ruby

RUN apt-get update && apt-get install -qq -y build-essential libpq-dev postgresql-client-9.4 apt-transport-https

# Using Debian, as root
RUN curl -sL https://deb.nodesource.com/setup_9.x | bash -
RUN apt-get install -y nodejs

ENV APP_HOME /usr/src/app
RUN mkdir $APP_HOME
WORKDIR $APP_HOME

ADD Gemfile* $APP_HOME/
ADD package.json* $APP_HOME/
RUN bundle install --jobs 20 --retry 5
RUN npm install
```

## Create Docker compose file

```
version: '3'

services:
  webapp:
    build: ./docker-workshop-app
    depends_on:
      - postgres
    environment:
      - RAILS_PORT=3000
      - DB_USERNAME=user
      - DB_PASSWORD=password
      - DB_HOST=postgres
    volumes:
      - './docker-workshop-app/:/usr/src/app'
    ports:
      - '3000:3000'
    networks:
      - workshop
    stdin_open: true
    tty: true
    working_dir: '/usr/src/app'
    command:
      - /bin/bash
      - -c
      - bundle install --jobs 20 --retry 5; npm install; rm ./tmp/pids/server.pid; rails db:setup db:seed; rails s

  postgres:
    image: postgres
    environment:
      - POSTGRES_USER=user
      - POSTGRES_PASSWORD=password
    networks:
      - workshop
    ports:
      - '5432:5432'
    volumes:
      - './postgres:/var/lib/postgresql/data'

networks:
  workshop:
    driver: bridge
```
